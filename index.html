}

public class TypeCasting {
    public static void main(String[] args) {

        // ── Widening (Implicit) — no data loss ───
        // byte → short → int → long → float → double
        int    i = 100;
        double d = i;          // int → double  (automatic)
        long   l = i;          // int → long    (automatic)
        System.out.println("Widening: " + i + " → " + d);

        // ── Narrowing (Explicit) — may lose data ─
        double x = 9.99;
        int    y = (int) x;    // double → int  (manual cast)
        System.out.println("Narrowing: " + x + " → " + y); // 9

        // ── String conversions ───────────────────
        int    num  = Integer.parseInt("123");     // String → int
        double dbl  = Double.parseDouble("3.14");  // String → double
        String str1 = String.valueOf(456);         // int → String
        String str2 = Integer.toString(789);       // int → String

        System.out.println(num + " " + dbl);
        System.out.println(str1 + " " + str2);
    }
}
⚠️
Narrowing risk — double 9.99 becomes int 9. Decimal part is lost, not rounded. Use carefully.

public class Operators {
    public static void main(String[] args) {
        int a = 10, b = 3;

        // ── Arithmetic ───────────────────────────
        System.out.println(a + b);   // 13  addition
        System.out.println(a - b);   // 7   subtraction
        System.out.println(a * b);   // 30  multiplication
        System.out.println(a / b);   // 3   division (integer)
        System.out.println(a % b);   // 1   modulus (remainder)

        // ── Increment / Decrement ────────────────
        int x = 5;
        System.out.println(x++);  // 5  (post: use then increment)
        System.out.println(++x);  // 7  (pre:  increment then use)

        // ── Relational ───────────────────────────
        System.out.println(a > b);   // true
        System.out.println(a < b);   // false
        System.out.println(a >= b);  // true
        System.out.println(a == b);  // false
        System.out.println(a != b);  // true

        // ── Logical ──────────────────────────────
        System.out.println(a > 5 && b < 5);  // true   (AND)
        System.out.println(a > 5 || b > 5);  // true   (OR)
        System.out.println(!(a > 5));         // false  (NOT)

        // ── Bitwise ──────────────────────────────
        System.out.println(a & b);   // 2   AND
        System.out.println(a | b);   // 11  OR
        System.out.println(a ^ b);   // 9   XOR
        System.out.println(~a);      // -11 NOT
        System.out.println(a << 1);  // 20  left shift  (×2)
        System.out.println(a >> 1);  // 5   right shift (÷2)

        // ── Ternary ──────────────────────────────
        int max = (a > b) ? a : b;   // if a>b then a else b
        System.out.println("Max: " + max); // 10

        // ── Assignment ───────────────────────────
        int n = 10;
        n += 5;   // n = n + 5  = 15
        n -= 3;   // n = n - 3  = 12
        n *= 2;   // n = n * 2  = 24
        n /= 4;   // n = n / 4  = 6
        n %= 4;   // n = n % 4  = 2
        System.out.println(n); // 2
    }
}
ℹ️
Ternary syntax — condition ? valueIfTrue : valueIfFalse. The condition MUST evaluate to boolean.

import java.util.Arrays;

public class ArraysDemo {
    public static void main(String[] args) {

        // ── Declare & Initialize ──────────────────
        int[] a = new int[5];           // all zeros
        int[] b = {10, 20, 30, 40, 50}; // inline init
        int[] c = new int[]{5, 3, 8, 1}; // alternative

        // ── Access & Modify ───────────────────────
        System.out.println(b[0]);    // 10  (0-indexed!)
        System.out.println(b[4]);    // 50
        System.out.println(b.length);// 5
        b[2] = 99;                   // modify element

        // ── Traverse ─────────────────────────────
        for (int val : b)
            System.out.print(val + " ");  // 10 20 99 40 50

        // ── Arrays utility methods ────────────────
        int[] arr = {5, 3, 8, 1, 9, 2};
        Arrays.sort(arr);                           // in-place sort
        System.out.println(Arrays.toString(arr));   // [1, 2, 3, 5, 8, 9]
        int idx = Arrays.binarySearch(arr, 8);      // 4 (after sort)
        System.out.println("Found 8 at: " + idx);

        // ── Find max & min ────────────────────────
        int max = arr[0], min = arr[0];
        for (int v : arr) {
            if (v > max) max = v;
            if (v < min) min = v;
        }
        System.out.println("Max=" + max + " Min=" + min);

        // ── 2D Array ─────────────────────────────
        int[][] matrix = {
            {1, 2, 3},
            {4, 5, 6},
            {7, 8, 9}
        };
        // Rows × Columns
        System.out.println("Rows: " + matrix.length);     // 3
        System.out.println("Cols: " + matrix[0].length);  // 3
        System.out.println("Center: " + matrix[1][1]);    // 5

        // Print entire matrix
        for (int[] row : matrix) {
            for (int v : row)
                System.out.printf("%3d", v);
            System.out.println();
        }

        // ── Copy arrays ───────────────────────────
        int[] copy = Arrays.copyOf(arr, arr.length);
        int[] partial = Arrays.copyOfRange(arr, 1, 4); // idx 1..3
        System.out.println(Arrays.toString(partial));
    }
}
⚠️
ArrayIndexOutOfBoundsException — happens when you access index ≥ length. Arrays are 0-indexed: valid range is 0 to length-1.

// CLASS = blueprint/template
public class Student {
    // Fields (attributes/state)
    String name;
    int    age;
    double gpa;

    // Method (behavior)
    public void display() {
        // "this" refers to current object
        System.out.println("Name: " + this.name +
                          " | Age: "  + this.age  +
                          " | GPA: "  + this.gpa);
    }

    public boolean isHonors() {
        return this.gpa >= 3.5;
    }
}

// OBJECT = instance of a class
public class Main {
    public static void main(String[] args) {

        Student s1 = new Student();  // create object
        s1.name = "Alice";           // set fields
        s1.age  = 20;
        s1.gpa  = 3.8;
        s1.display();                // call method

        Student s2 = new Student();
        s2.name = "Bob";
        s2.age  = 22;
        s2.gpa  = 3.2;
        s2.display();

        System.out.println(s1.name + " honors: " + s1.isHonors()); // true
        System.out.println(s2.name + " honors: " + s2.isHonors()); // false
    }
}

// Methods are shared across all objects

public class Person {
    String name;
    int    age;
    String city;

    // ── Default Constructor ───────────────────
    public Person() {
        this("Unknown", 0, "Unknown"); // constructor chaining ↓
    }

    // ── Parameterized Constructor ─────────────
    public Person(String name, int age) {
        this(name, age, "Unknown");    // constructor chaining ↓
    }

    // ── Full Parameterized Constructor ────────
    public Person(String name, int age, String city) {
        this.name = name;
        this.age  = age;
        this.city = city;
    }

    // ── Copy Constructor ─────────────────────
    public Person(Person other) {
        this.name = other.name;
        this.age  = other.age;
        this.city = other.city;
    }

    public void display() {
        System.out.println(name + " | " + age + " | " + city);
    }

    public static void main(String[] args) {
        Person p1 = new Person();                     // default
        Person p2 = new Person("Alice", 25);          // 2-param
        Person p3 = new Person("Bob", 30, "NYC");     // full
        Person p4 = new Person(p3);                   // copy of p3

        p1.display(); // Unknown | 0 | Unknown
        p2.display(); // Alice   | 25 | Unknown
        p3.display(); // Bob     | 30 | NYC
        p4.display(); // Bob     | 30 | NYC  (copy)
    }
}

Inheritance
extends · super · single · multilevel · hierarchical
▼
// ── Parent / Base class ───────────────────────
class Animal {
    String name;
    int    age;

    public Animal(String name, int age) {
        this.name = name;
        this.age  = age;
    }

    public void eat()   { System.out.println(name + " is eating"); }
    public void sleep() { System.out.println(name + " is sleeping"); }
    public void info()  { System.out.println(name + ", age " + age); }
}

// ── Child / Derived class ─────────────────────
class Dog extends Animal {
    String breed;

    public Dog(String name, int age, String breed) {
        super(name, age);  // ← calls Animal constructor (MUST be first)
        this.breed = breed;
    }

    public void bark() { System.out.println(name + " says: Woof!"); }

    @Override
    public void info() {
        super.info();      // call parent method
        System.out.println("Breed: " + breed);
    }
}

// ── Grandchild class (Multilevel) ────────────
class Puppy extends Dog {
    public Puppy(String name) {
        super(name, 0, "Mixed");
    }
    public void play() { System.out.println(name + " is playing!"); }
}

public class Main {
    public static void main(String[] args) {
        Dog d = new Dog("Rex", 3, "Labrador");
        d.eat();    // from Animal ✓
        d.bark();   // from Dog ✓
        d.info();   // overridden (calls super too) ✓

        Puppy p = new Puppy("Max");
        p.eat();    // from Animal ✓ (grandparent)
        p.bark();   // from Dog ✓   (parent)
        p.play();   // from Puppy ✓ (own)
    }
}
// Java only supports SINGLE inheritance for classes
// Multiple inheritance achieved via Interfaces
⚠️
super() rule — super() must be the FIRST statement in a child constructor. Java auto-inserts super() if you don't write it (calls no-arg parent constructor).

// METHOD OVERLOADING — same name, different parameters
// Resolved at COMPILE TIME (static dispatch)
class Calculator {
    // Different number of params
    public int add(int a, int b)         { return a + b; }
    public int add(int a, int b, int c)  { return a + b + c; }

    // Different parameter types
    public double add(double a, double b){ return a + b; }
    public String add(String a, String b){ return a + b; }

    // Different order of types
    public String concat(int n, String s){ return n + s; }
    public String concat(String s, int n){ return s + n; }
}

// Main
Calculator c = new Calculator();
System.out.println(c.add(2, 3));          // 5      int version
System.out.println(c.add(2.5, 3.5));      // 6.0    double version
System.out.println(c.add(1, 2, 3));       // 6      3-param version
System.out.println(c.add("Hello", "!"));  // Hello! String version
// METHOD OVERRIDING — same name + params, different class
// Resolved at RUNTIME (dynamic dispatch) — key to polymorphism
class Shape {
    public void draw()        { System.out.println("Drawing shape"); }
    public double area()      { return 0; }
}

class Circle extends Shape {
    double r;
    Circle(double r) { this.r = r; }

    @Override                  // ← annotation (recommended)
    public void draw()   { System.out.println("Drawing Circle r=" + r); }
    @Override
    public double area() { return Math.PI * r * r; }
}

class Rectangle extends Shape {
    double l, w;
    Rectangle(double l, double w) { this.l = l; this.w = w; }

    @Override
    public void draw()   { System.out.println("Drawing Rect " + l + "x" + w); }
    @Override
    public double area() { return l * w; }
}

// ── Runtime Polymorphism (upcasting) ─────────
Shape[] shapes = { new Circle(5), new Rectangle(4, 6), new Shape() };
for (Shape s : shapes) {
    s.draw();                       // calls correct version at runtime!
    System.out.printf("Area: %.2f%n", s.area());
}
// Output:
// Drawing Circle r=5.0    Area: 78.54
// Drawing Rect 4.0x6.0   Area: 24.00
// Drawing shape           Area: 0.00

// Abstract class — cannot be instantiated directly
// Can have both abstract AND concrete methods
abstract class Vehicle {
    String brand;
    int    speed;

    public Vehicle(String brand, int speed) {
        this.brand = brand;
        this.speed = speed;
    }

    // ABSTRACT — subclass MUST implement this
    public abstract void start();
    public abstract String fuelType();

    // CONCRETE — subclass inherits as-is
    public void stop() {
        System.out.println(brand + " stopped.");
    }
    public void displayInfo() {
        System.out.println(brand + " | " + speed + " kmph | " + fuelType());
    }
}

    @Override

    @Override

}

// Interface — pure abstraction contract
// All methods public abstract by default (before Java 8)
// Supports MULTIPLE IMPLEMENTATION

interface Flyable {
    int MAX_ALTITUDE = 10000; // implicitly public static final
    void fly();               // implicitly public abstract
    void land();
}

interface Swimmable {
    void swim();
}

// Default method (Java 8+) — optional to override
interface Printable {
    void print();
    default void preview() {
        System.out.println("Preview: " + getClass().getSimpleName());

}

// Class can implement MULTIPLE interfaces
class Duck implements Flyable, Swimmable {

    @Override public void swim() { System.out.println(name + " swims!"); }
}

// Interface can extend another interface
interface Aquatic extends Swimmable {
    void dive();
}

Duck d = new Duck("Donald");
d.fly(); d.swim();
System.out.println("Max alt: " + Flyable.MAX_ALTITUDE);

public class BankAccount {
    // PRIVATE fields — cannot be accessed directly from outside
    private String owner;
    private double balance;
    private String accNumber;

    public BankAccount(String owner, String accNumber, double balance) {
        this.owner     = owner;
        this.accNumber = accNumber;
        this.balance   = (balance >= 0) ? balance : 0;
    }

    // GETTERS — read-only access
    public String getOwner()     { return owner; }
    public double getBalance()   { return balance; }
    public String getAccNumber() { return "****" + accNumber.substring(4); }

    // SETTERS with VALIDATION
    public void setOwner(String owner) {
        if (owner != null && !owner.isBlank())
            this.owner = owner;
    }

    // Business logic methods (no raw setter for balance)
    public void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            System.out.printf("Deposited %.2f. Balance: %.2f%n", amount, balance);
        } else {
            System.out.println("❌ Deposit amount must be positive");
        }
    }

    public void withdraw(double amount) {
        if (amount <= 0)
            System.out.println("❌ Invalid amount");
        else if (amount > balance)
            System.out.println("❌ Insufficient balance");
        else {
            balance -= amount;
            System.out.printf("Withdrawn %.2f. Balance: %.2f%n", amount, balance);
        }
    }
}

    public static String getCompany()  { return companyName; }

    // Instance method — can access both static and instance
    public void display() {
        System.out.println("ID: " + empId + " | " + name +
                          " | Company: " + companyName);
    }
}

public static void fibonacci(int n) {
    int a = 0, b = 1;
    for (int i = 0; i < n; i++) {
        System.out.print(a + " ");
        int c = a + b; a = b; b = c;
    }
}

// ── Fibonacci Recursive ───────────────────────
public static int fibRec(int n) {
    if (n <= 1) return n;
    return fibRec(n-1) + fibRec(n-2);
}

// ── Factorial Recursive ───────────────────────
public static long factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

// ── Prime Check ───────────────────────────────
public static boolean isPrime(int n) {
    if (n < 2) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (int i = 3; i <= Math.sqrt(n); i += 2)
        if (n % i == 0) return false;
    return true;
}
// isPrime(17) → true, isPrime(20) → false

// ── Number Palindrome ─────────────────────────
public static boolean isNumPalindrome(int n) {
    int orig = n, rev = 0;
    while (n > 0) {
        rev = rev * 10 + n % 10;
        n /= 10;
    }
    return orig == rev;
}
// isNumPalindrome(121) → true

// ── Armstrong Number ──────────────────────────
public static boolean isArmstrong(int n) {
    int orig = n, sum = 0;
    int digits = String.valueOf(n).length();
    while (n > 0) {
        sum += (int) Math.pow(n % 10, digits);
        n /= 10;
    }
    return sum == orig;
}
// isArmstrong(153) → true  (1³+5³+3³=153)
// isArmstrong(370) → true  (3³+7³+0³=370)

public static boolean isPermutation(String s1, String s2) {
    if (s1.length() != s2.length()) return false;
    int[] count = new int[256];
    for (char c : s1.toCharArray()) count[c]++;
    for (char c : s2.toCharArray())
        if (--count[c] < 0) return false;

// ── Bubble Sort ───────────────────────────────
// O(n²) — compare adjacent, swap if wrong order
public static void bubbleSort(int[] arr) {
    int n = arr.length;
    for (int i = 0; i < n - 1; i++) {
        boolean swapped = false;
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                int tmp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = tmp;
                swapped = true;
            }
        }
        if (!swapped) break; // already sorted optimization
    }
}

// ── Selection Sort ────────────────────────────
// O(n²) — find min, put it at correct position
public static void selectionSort(int[] arr) {
    for (int i = 0; i < arr.length - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < arr.length; j++)
            if (arr[j] < arr[minIdx]) minIdx = j;
        int tmp = arr[minIdx]; arr[minIdx] = arr[i]; arr[i] = tmp;
    }
}

// ── Insertion Sort ────────────────────────────
// O(n²) avg, O(n) best — insert element in correct position
public static void insertionSort(int[] arr) {
    for (int i = 1; i < arr.length; i++) {
        int key = arr[i], j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j]; j--;
        }
        arr[j + 1] = key;
    }
}

// ── Linear Search ────────────────────────────
public static int linearSearch(int[] arr, int target) {
    for (int i = 0; i < arr.length; i++)
        if (arr[i] == target) return i;
    return -1;
}

// ── Binary Search (array must be sorted) ─────
public static int binarySearch(int[] arr, int target) {
    int lo = 0, hi = arr.length - 1;
    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        if      (arr[mid] == target) return mid;
        else if (arr[mid] < target)  lo = mid + 1;
        else                         hi = mid - 1;
    }
    return -1;
}
// int[] a = {1,3,5,7,9,11}; binarySearch(a, 7) → 3
